- [ ] **Step 1: 提取 CSV 基础统计**

```powershell
$csv = Import-Csv "D:\Repositories\QT-Resources\RTSP-Pusher\bin\MSVC2017_x86_Release\csv_baseline.csv"

$totalRows = $csv.Count
$firstTs = [double]$csv[0].timestamp
$lastTs = [double]$csv[-1].timestamp
$durationSec = $lastTs - $firstTs

# 稳定段：跳过前 2 行和后 2 行
$stable = $csv[2..($csv.Count - 3)]

# 视频采集 fps（每 5 秒窗口的增量 / 5）
$capturedDiffs = for ($i = 1; $i -lt $csv.Count; $i++) {
    [double]$csv[$i].videoCaptured - [double]$csv[$i-1].videoCaptured
}
$avgCaptureFps = ($capturedDiffs | Measure-Object -Average).Average / 5

# 编码 fps
$encodedDiffs = for ($i = 1; $i -lt $csv.Count; $i++) {
    [double]$csv[$i].videoEncoded - [double]$csv[$i-1].videoEncoded
}
$avgEncodeFps = ($encodedDiffs | Measure-Object -Average).Average / 5

# 稳定段统计
$stableCaptureFps = ($capturedDiffs[2..($capturedDiffs.Count-3)] | Measure-Object -Average).Average / 5
$stableEncodeFps = ($encodedDiffs[2..($encodedDiffs.Count-3)] | Measure-Object -Average).Average / 5

Write-Host "总时长: $([math]::Round($durationSec, 1)) 秒"
Write-Host "平均采集fps: $([math]::Round($avgCaptureFps, 2))"
Write-Host "平均编码fps: $([math]::Round($avgEncodeFps, 2))"
Write-Host "稳定段采集fps: $([math]::Round($stableCaptureFps, 2))"
Write-Host "稳定段编码fps: $([math]::Round($stableEncodeFps, 2))"
```

Expected: 时长 ≈ 600s，fps 接近 30
